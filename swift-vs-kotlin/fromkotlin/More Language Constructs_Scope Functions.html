                  fun main() {
    val str = "Hello"

    str.run {
        println("The receiver string length: $length")
        //println("The receiver string length: ${this.length}") // does the same

​

                  fun displaySubstringPosition(input: String, sub: String) {

        println("The substring $sub is found in $input.")

    }
